===========================|__clz32()__|============================
...用于计算32位二进制的前置0的个数；一般结合位移运算使用。
   Math.clz32();	//32;
   Math.clz32(0);	//32;
   Math.clz32(1);	//31;
   Math.clz32(11);	//28;
   Math.clz32(0b11);	//30;
   Math.clz32(1 << 1);	//30;

   Math.sinh(x);	//返回x的双曲正弦

   Math.tanh(x);	//返回x的双曲正切
   Math.asinh(x);	//返回x的反双曲正弦
   Math.acosh(x);	//返回x的反双曲余弦
   Math.atanh(x);	//返回x的反双曲正切
