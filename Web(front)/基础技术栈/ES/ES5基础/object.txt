#对象的键名不仅可以动态定义,
#还可以使用 中文名,数字,所有的特殊符号,
#但是含有符号的键名需要使用冒号来确定其类型是字符串
例如:
{中国:"china"}	//正确;
{中-国:"china"}	//报错
{"中-国":"china"}	//正确
{1:"1"}	//正确
{0xaa:"170"}	//正确 输出的结果:{170:"170"}

var key = "12344556";
{key:"123"}	//输出的结果:{key:"123"}


#如果想要动态定义键名,可以使用for in就能实现
例如:
	for(var item in object){
	var newObj = {};
	newObj[item] = object[item];
	}
