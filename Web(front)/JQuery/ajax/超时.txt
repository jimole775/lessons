	var ajaxInstance = $.ajax({
			type: "POST",
			timeout: 10000, //超时时间设置，单位毫秒
			complete: function (XMLHttpRequest, status) {
				switch (status) {
					case "timeout":
						ajaxInstance.abort();	//超时之后必须abort当前请求
						that.jsRecvServerData("timeout", "服务器请求超时", callback);
						break;
				}
			}
		});
