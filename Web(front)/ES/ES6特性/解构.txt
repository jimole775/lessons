
var [a,b] = ["a","b"];
//a = "a",b = "b"


var [a,b] = {a:"a",b:"b"};
//a = "a",b = "b";

###解构别名：
var {a:b} = {a:"b"}
//b = "b"

###解构时设置默认值
var {b=1} = {a:"b"}
//b = 1;
